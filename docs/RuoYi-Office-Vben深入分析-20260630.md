4. 执行数据迁移
5. 数据验证

**迁移脚本示例**：
```python
import sqlite3
import mysql.connector

# SQLite → MySQL迁移
sqlite_conn = sqlite3.connect('database/project_management.db')
mysql_conn = mysql.connector.connect(
    host='localhost',
    user='root',
    password='***',
    database='ruoyi_office_vben'
)

# 迁移contracts表
sqlite_cursor = sqlite_conn.cursor()
mysql_cursor = mysql_conn.cursor()

sqlite_cursor.execute('SELECT * FROM contracts')
for row in sqlite_cursor.fetchall():
    # 映射到RuoYi的合同表结构
    mysql_cursor.execute(
        'INSERT INTO sys_contract (...) VALUES (...)',
        row
    )

mysql_conn.commit()
```

services:
  mysql:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: ***
      MYSQL_DATABASE: ruoyi_office_vben
    volumes:
      - ./mysql-data:/var/lib/mysql

  redis:
    image: redis:latest

  nacos:
    image: nacos/nacos-server:latest
    environment:
      MODE: standalone

  ruoyi-gateway:
    image: ruoyi/ruoyi-gateway:latest
    ports:
      - "8080:8080"
    depends_on:
      - mysql
      - redis
      - nacos
