def identify_project_region(project_name):
    """
    识别项目区域（四川 vs 省外）
    
    返回：
    - 'sichuan': 四川项目
    - 'other': 省外项目
    """
    sichuan_keywords = ['成都', '绵阳', '德阳', '乐山', '凉山', '阿坝', 
                        '雅安', '眉山', '自贡', '泸州', '宜宾', '达州', 
                        '广元', '遂宁', '内江', '南充', '广安', '巴中', 
                        '资阳', '攀枝花']
    
    other_keywords = ['重庆', '云南', '贵州', '陕西', '湖北', '湖南', 
                      '河南', '甘肃', '青海', '西藏']
    
    for keyword in sichuan_keywords:
        if keyword in project_name:
            return 'sichuan'
    
    for keyword in other_keywords:
        if keyword in project_name:
            return 'other'
    
    return 'sichuan'  # 默认为四川项目

### 步骤4：生成飞书表格更新格式

```python
def generate_feishu_format(records, total_amount, record_type='invoiced'):
    """
    生成飞书表格标准格式
    
    参数：
    - records: 记录列表
    - total_amount: 总金额
    - record_type: 'invoiced' 或 'payment'
    
    返回：
    - formatted_text: 标准格式文本
    """
    lines = []
    
    # 1. 添加明细记录
    for record in records:
        date = record['date']
        amount = record['amount']
        lines.append(f"{date}：{amount}元")
    
    # 2. 添加汇总行
    if record_type == 'invoiced':
        lines.append(f"已开：{total_amount}元")
    else:
        lines.append(f"已回：{total_amount}元")
    
    formatted_text = '\n'.join(lines)
    return formatted_text
```

## ⚠️ 常见问题与解决方案

### 问题1：财务表格显示"完结"，但提取为0

**原因**：没有识别到明细记录

**解决方案**：
- 检查是否有明细记录
- 如果没有明细记录，使用合同金额作为已回款金额

---

### 问题2：省外项目混入HTML统计

**原因**：未识别项目区域

**解决方案**：
- 添加项目区域识别步骤
- 省外项目单独处理

---

### 问题3：汇总金额与明细总和不一致

**原因**：明细记录提取遗漏或重复

**解决方案**：
- 检查明细记录提取逻辑
- 去重算法优化
