def extract_contract_amount(value):

        return float(value) * 10000 if pd.notna(value) else 0
    except:
        return 0  # 异常格式跳过（如"6..2153"）

# 2. 已开票提取（支持多种格式）
def extract_invoiced_amount(text):
    """
    支持格式：
    - "已开完：318000元 (2024.9.24)\n254336.4元 (2024.10.17)"
    - "已开：290000元\n170000元 (2025.11.24)"
    - "已开完元"（空金额）
    """
    amounts = re.findall(r'(\d+\.?\d*)\s*元', str(text))
    return sum([float(a) for a in amounts]) if amounts else 0

# 3. 回款进度提取（支持多种格式）
def extract_payment_amount(text):
    """
    支持格式：
    - "已回：674400元\n180300元（2024.10.18）"
    - "完结"
    - "已回完\n226480元（26-1-14）"
    """
    amounts = re.findall(r'(\d+\.?\d*)\s*元', str(text))
    return sum([float(a) for a in amounts]) if amounts else 0

### 步骤3：数据对比与差异分析

**对比逻辑**：

```python
def compare_projects(financial_data, feishu_data):
    """
    对比规则：
    1. 项目名称匹配（模糊匹配，忽略"合同"后缀）
    2. 已开票金额对比（财务表格 vs 飞书表格）
    3. 已回款金额对比（财务表格 vs 飞书表格）
    4. 列出差异项目
    """
    differences = []

    for fin_project, fin_data in financial_data.items():
        # 匹配飞书项目
        feishu_match = match_project_name(fin_project, feishu_data)

        if feishu_match:
            # 对比已开票
            if fin_data['invoiced'] != feishu_match['invoiced']:
                differences.append({
                    'project': fin_project,
                    'field': '已开票',
                    'financial': fin_data['invoiced'],
                    'feishu': feishu_match['invoiced'],
                    'difference': fin_data['invoiced'] - feishu_match['invoiced']
                })

            # 对比已回款
            if fin_data['payment'] != feishu_match['payment']:
                differences.append({
                    'project': fin_project,
                    'field': '已回款',
                    'financial': fin_data['payment'],
                    'feishu': feishu_match['payment'],
                    'difference': fin_data['payment'] - feishu_match['payment']
                })

    return differences
```

## 📊 格式识别与转换规则

### 已开票字段格式识别

| 财务表格格式 | 提取方法 | 转换为飞书格式 |
|-------------|---------|--------------|
| `已开完：318000元 (2024.9.24)` | 提取金额：318000 + 254336.4 | `2024-09-24：318000元\n2024-10-17：254336.4元\n已开完` |
| `已开：290000元` | 提取金额：290000 | `已开：290000元\n未开：195347元` |

|-------------|---------|--------------|

| `完结` | 提取金额：从上下文推断 | `已完结` |

## ⚠️ 特殊情况处理

### 情况1：项目名称不匹配

**处理方法**：
- 模糊匹配（忽略"合同"、"技术服务"等后缀）
- 手动确认（列出无法匹配的项目）
- 跳过更新（如果确实无法匹配）

### 情况2：数据格式异常

**处理方法**：
- 合同金额异常（如"6..2153"）→ 跳过或设为0
- 开票金额重复计算 → 去重
- 日期格式混乱 → 标准化为YYYY-MM-DD

### 情况3：飞书表格数据缺失

**处理方法**：
- 项目状态为"商机跟踪中" → 跳过更新
- 项目金额为0 → 确认后更新
- 字段为空 → 从财务表格补充
