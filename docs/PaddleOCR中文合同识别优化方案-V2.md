### 3.2 后处理纠错引擎

#### 纠错规则1：字符混淆修正

```python
CHAR_CONFUSION_MAP = {
    # 数字-字母混淆
    '0': 'O', 'O': '0',
    '1': 'I', 'I': '1', 'l': '1',
    '5': 'S', 'S': '5',
    '8': 'B', 'B': '8',
    
    # 中文相似字
    '网': '钢', '钢': '网',
    '己': '已', '已': '己',
    '未': '末', '末': '未',
}

def correct_contract_number(text: str) -> str:
    """修正合同编号中的字符混淆"""
    # 合同编号格式：SGSCCD00JYJS2401318
    # 规则：前缀字母 + 数字 + 字母 + 数字
    
    # 提取合同编号
    match = re.search(r'([A-Z]{6})(\d{2})([A-Z]{4})(\d{7})', text)
    if match:
        prefix = match.group(1)
        num1 = match.group(2)
        letters = match.group(3)
        num2 = match.group(4)
        
        # 修正：num1和num2必须全是数字
        num1 = num1.replace('O', '0').replace('I', '1')
        num2 = num2.replace('O', '0').replace('I', '1')
        
        # 修正：letters必须全是字母
        letters = letters.replace('0', 'O').replace('1', 'I')
        
        return f"{prefix}{num1}{letters}{num2}"
    
    return text
```

#### 纠错规则2：日期格式标准化

```python
def correct_date(text: str) -> str:
    """修正日期格式错误"""
    # 常见错误：2024.8.13 → 204.8.13
    # 规则：年份必须是4位数
    
    patterns = [
        (r'(\d{2})(\d{2})\.(\d{1,2})\.(\d{1,2})', r'\1\2-\3-\4'),  # 2024.8.13 → 2024-8-13
        (r'(\d{4})\.(\d{1,2})\.(\d{1,2})', r'\1-\2-\3'),  # 2024.8.13 → 2024-8-13
    ]
    
    for pattern, replacement in patterns:
        text = re.sub(pattern, replacement, text)
    
    return text
```

```python
def extract_amount_info(text: str) -> Dict:
    """增强金额信息提取（支持大写+小写）"""
    
    info = {}
    
    # 1. 从"项目支付计划"章节提取（第12页）
    if '项目支付计划' in text or '项目经费总额' in text:
        # 总金额（含税）- 支持大写+小写
        match = re.search(
            r'项目经费总额为人民币（大写）[^\n]*\n（￥([\d,\.]+)元）',
            text, re.MULTILINE
        )
        if match:
            info['合同总金额（含税）'] = match.group(1).replace(',', '')
        
        # 不含税金额
        match = re.search(
            r'不含税价\s*人民币（￥([\d,\.]+)元）',
            text
        )
        if match:
            info['不含税金额'] = match.group(1).replace(',', '')
        
        # 税率
        match = re.search(r'增值税税率\s*(\d+)%', text)
        if match:
            info['税率'] = match.group(1) + '%'
        
        # 税额
        match = re.search(r'增值税税额\s*([\d,\.]+)\s*元', text)
        if match:
            info['税额'] = match.group(1).replace(',', '')
    
    # 2. 从项目简表提取（第3页）
    if not info:
        match = re.search(r'合同总\s*金额（万元）\s*\n[\s\S]*?(\d+\.?\d*)\s*\n', text)
        if match:
            amount_wan = float(match.group(1))
            info['合同总金额（含税）'] = str(amount_wan * 10000)
    
    return info if info else None
```

#### 优化5：项目名称合并提取

**当前规则**：

**优化后规则**：
```python
def extract_project_name(text: str, page: int = 1) -> str:
    """合并提取项目名称（支持换行）"""
    
    if page != 1:
        return None
    
    # 查找"项目名称："开头的行
    lines = text.split('\n')
    project_lines = []
    capturing = False
    
    for line in lines:
        if '项目名称[：:]' in line:
            capturing = True
            project_lines.append(line.split('项目名称[：:]')[1].strip())
        elif capturing:
            # 继续捕获下一行（直到遇到下一个字段）
            if re.match(r'[^\n]{10,50}[：:]', line):
                break
            project_lines.append(line.strip())
    
    if project_lines:
        return ''.join(project_lines)
    
    return None
```

#### 优化7：交付物章节限定提取

**当前规则**：
```python
pattern_numbered = r'[（(①②③④⑤⑥⑦⑧⑨⑩]?\d+[）)]?[：:、]?\s*[^\n]+'
```

**优化后规则**：
```python
def extract_deliverables(text: str) -> List[str]:
    """章节限定提取交付物清单"""
    
    deliverables = []
    
    # 1. 定位"4.提供的最终成果"章节
    chapter_start = text.find('4.提供的最终成果')
    chapter_end = text.find('5.进度计划内容及考核目标')
    
    if chapter_start > 0 and chapter_end > chapter_start:
        chapter_text = text[chapter_start:chapter_end]
        
        # 2. 提取章节内的编号项
        pattern = r'[（(]\d+[）)][：:]?\s*[^\n]+'
        matches = re.findall(pattern, chapter_text)
        
        # 3. 过滤关键字
        keywords = ['论文', '专利', '软件著作权', '技术报告', '系统', '平台', '策略']
        for match in matches:
            if any(kw in match for kw in keywords):
                deliverables.append(match.strip())
    
    return deliverables
```
