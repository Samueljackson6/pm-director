# OCR纠错规则库架构设计

> **设计日期**：2026-06-24  
> **目标**：建立通用纠错规则库，支持不同类型合同

## 一、架构概述

### 1.1 核心思路

**从"硬编码规则"升级为"规则库系统"**

| 维度 | 当前方案 | 规则库方案 |
|------|---------|-----------|
| **规则管理** | 硬编码在代码中 | JSON配置文件 |
| **规则更新** | 需改代码 | 只改配置 |
| **规则扩展** | 需改代码 | 新增配置文件 |
| **多类型支持** | 不支持 | 支持不同合同类型 |

## 三、规则管理器

### 3.1 RuleManager类设计

```python
class RuleManager:
    """规则管理器"""
    
    def __init__(self, rules_dir: str = "correction_rules"):
        """
        初始化规则管理器
        
        Args:
            rules_dir: 规则文件目录
        """
        self.rules_dir = rules_dir
        self.rules_cache = {}  # 规则缓存
        self.load_all_rules()
    
    def load_all_rules(self):
        """加载所有规则文件"""
        for rule_file in Path(self.rules_dir).glob("*.json"):
            rule_name = rule_file.stem
            self.rules_cache[rule_name] = json.load(open(rule_file))
    
    def get_correction_rule(self, rule_type: str, contract_type: str = None):
        """
        获取纠错规则
        
        Args:
            rule_type: 规则类型（如'char_confusion', 'contract_number'）
            contract_type: 合同类型（如'tech_project'）
        """
        if contract_type and contract_type in self.rules_cache.get('contract_types', {}):
            return self.rules_cache['contract_types'][contract_type].get(rule_type)
        
        return self.rules_cache.get(rule_type)
    
    def apply_char_correction(self, text: str, context: str):
        """
        应用字符混淆纠错
        
        Args:
            text: 待纠错文本
            context: 上下文类型（如'contract_number', 'amount', 'company_name')
        """
        char_rules = self.get_correction_rule('char_confusion')
        
        corrected_text = text
        for rule in char_rules['rules']:
            for mapping in rule['mappings']:
                if mapping['context'] == context or mapping['context'] == 'any':
                    # 根据上下文判断应该纠正为哪个字符
                    if context in ['contract_number', 'amount']:
                        # 数字场景：字母→数字
                        if mapping['ocr'].isalpha():
                            corrected_text = corrected_text.replace(
                                mapping['ocr'], 
                                mapping['correct'][0] if mapping['correct'][0].isdigit() else mapping['ocr']
                            )
                    elif context == 'company_name':
                        # 公司名称场景：应用中文纠错
                        corrected_text = corrected_text.replace(
                            mapping['ocr'], 
                            mapping['correct']
                        )
        
        return corrected_text
    
    def validate_contract_number(self, number: str, contract_type: str):
        """
        验证并纠错合同编号
        
        Args:
            number: 合同编号
            contract_type: 合同类型
        """
        contract_rules = self.get_correction_rule('contract_number', contract_type)
        
        if not contract_rules:
            return number
        
        # 验证格式
        if not re.match(contract_rules['pattern'], number):
            # 尝试纠错
            corrected = self.correct_contract_number(number, contract_rules)
            return corrected
        
        return number
    
    def correct_contract_number(self, number: str, rules: dict):
        """
        纠错合同编号
        
        Args:
            number: 合同编号
            rules: 合同编号规则
        """
        corrected = number
        
        # 应用位置规则
        for pos_rule in rules['position_rules']:
            position = pos_rule['position']
            expected_type = pos_rule['type']
            
            if '-' in position:
                start, end = map(int, position.split('-'))
                for i in range(start, end + 1):
                    if i < len(corrected):
                        char = corrected[i]
                        if expected_type == 'digit' and char.isalpha():
                            # 字母→数字
                            corrected = corrected[:i] + self.letter_to_digit(char) + corrected[i+1:]
                        elif expected_type == 'letter' and char.isdigit():
                            # 数字→字母
                            corrected = corrected[:i] + self.digit_to_letter(char) + corrected[i+1:]
        
        return corrected
    
    def letter_to_digit(self, letter: str):
        """字母转数字映射"""
        mapping = {'O': '0', 'o': '0', 'I': '1', 'i': '1', 'l': '1', 'S': '5', 's': '5', 'B': '8', 'b': '8'}
        return mapping.get(letter, letter)
    
    def digit_to_letter(self, digit: str):
        """数字转字母映射"""
        mapping = {'0': 'O', '1': 'I', '5': 'S', '8': 'B'}
        return mapping.get(digit, digit)
    
    def add_rule(self, rule_type: str, rule_data: dict, contract_type: str = None):
        """
        新增纠错规则
        
        Args:
            rule_type: 规则类型
            rule_data: 规则数据
            contract_type: 合同类型（可选）
        """
        if contract_type:
            # 新增合同类型规则
            contract_rules_file = Path(self.rules_dir) / 'contract_types' / f'{contract_type}.json'
            if contract_rules_file.exists():
                rules = json.load(open(contract_rules_file))
            else:
                rules = {"name": f"{contract_type}合同纠错规则", "version": "1.0"}
            
            rules[rule_type] = rule_data
            
            with open(contract_rules_file, 'w') as f:
                json.dump(rules, f, ensure_ascii=False, indent=2)
        else:
            # 新增通用规则
            rule_file = Path(self.rules_dir) / f'{rule_type}.json'
            with open(rule_file, 'w') as f:
                json.dump(rule_data, f, ensure_ascii=False, indent=2)
        
        # 更新缓存
        self.load_all_rules()
```

## 五、扩展方案

### 5.1 自动学习机制

**目标**：从识别结果自动学习纠错规则

```python
class RuleLearner:
    """规则学习器"""
    
    def learn_from_samples(self, samples: List[dict]):
        """
        从样本中学习纠错规则
        
        Args:
            samples: 样本数据（OCR结果 + 正确答案）
        
        Example:
            samples = [
                {
                    "ocr": "SGSCCD0OJYIS2401318",
                    "correct": "SGSCCD00JYJS2401318",
                    "type": "tech_project"
                }
            ]
        """
        # 分析差异
        for sample in samples:
            ocr = sample['ocr']
            correct = sample['correct']
            
            # 字符差异分析
            diff = self.analyze_diff(ocr, correct)
            
            # 生成规则
            rule = self.generate_rule_from_diff(diff, sample['type'])
            
            # 添加到规则库
            self.rule_manager.add_rule('contract_number', rule, sample['type'])
    
    def analyze_diff(self, ocr: str, correct: str):
        """分析OCR结果与正确答案的差异"""
        diffs = []
        for i in range(min(len(ocr), len(correct))):
            if ocr[i] != correct[i]:
                diffs.append({
                    "position": i,
                    "ocr_char": ocr[i],
                    "correct_char": correct[i]
                })
        return diffs
    
    def generate_rule_from_diff(self, diff: List[dict], contract_type: str):
        """根据差异生成规则"""
        position_rules = []
        for d in diff:
            position_rules.append({
                "position": str(d['position']),
                "ocr": d['ocr_char'],
                "correct": d['correct_char'],
                "confidence": 0.8  # 单样本置信度较低
            })
        
        return {
            "position_rules": position_rules,
            "source": "auto_learned",
            "samples_count": 1
        }
```

### 5.2 规则版本管理

**目标**：支持规则版本更新和回滚

```python
class RuleVersionManager:
    """规则版本管理器"""
    
    def commit_rule(self, rule_type: str, rule_data: dict):
        """提交规则变更"""
        # 保存当前版本
        current_version = self.get_current_version(rule_type)
        
        # 生成新版本号
        new_version = f"{rule_type}_v{current_version + 1}"
        
        # 保存新版本
        version_file = Path(self.rules_dir) / 'versions' / f'{new_version}.json'
        with open(version_file, 'w') as f:
            json.dump({
                "version": new_version,
                "timestamp": datetime.now().isoformat(),
                "rule": rule_data
            }, f, ensure_ascii=False, indent=2)
    
    def rollback(self, rule_type: str, target_version: int):
        """回滚到指定版本"""
        version_file = Path(self.rules_dir) / 'versions' / f'{rule_type}_v{target_version}.json'
        if version_file.exists():
            rule_data = json.load(open(version_file))
            self.rule_manager.add_rule(rule_type, rule_data['rule'])
```
