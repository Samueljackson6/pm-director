```sql
-- 付款总额校验触发器
CREATE TRIGGER validate_payment_total
AFTER INSERT OR UPDATE ON payments
WHEN (
  SELECT SUM(planned_amount) - contract_amount
  FROM payments p JOIN contracts c ON p.contract_id = c.contract_id
  WHERE p.contract_id = NEW.contract_id
) > contract_amount * 0.05  -- 允许5%误差
BEGIN
  SELECT RAISE(IGNORE);  -- 警告但不阻止
END;
```
